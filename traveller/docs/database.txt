


create table sector (id int auto_increment not null, name varchar(250) not null, x int not null, y int not null, primary key(id));

create table system (id int auto_increment not null, sector_id int not null, x int not null, y int not null, 
                     name varchar(250) not null, allegiance varchar(4) default 'Un', zone varchar(16) default 'Green',
                     base varchar(8), primary key(id));
create index sys1 on system (id);
create index sys2 on system (sector_id);
create index sys3 on system (sector_id, x, y);
                     
create table star (id int auto_increment not null, name varchar(250), system_id int not null, parent_id int not null, distance int not null,
                   form varchar(16), class varchar(16), type varchar(16),
                   primary key(id));
                   
create table planet (id int auto_increment not null, system_id int not null, parent_id int not null, moon boolean default false,
					name varchar(250), distance int not null, radius int not null, type varchar(32) not null,
					starport varchar(8), atmosphere varchar(32), pressure varchar(32), hydrographics int,
					population bigint, government varchar(32), law int, tech int, temperature varchar(32), life varchar(32),
					base varchar(8), trade varchar(64), day int default 86400,
					primary key(id));
create index planet1 on planet (system_id);

create table allegiance(id int not null auto_increment, code varchar(4) not null, name varchar(240) not null, 
                        colour varchar(12) default '#000000', primary key (id));
create index algnce1_idx on allegiance (id);
create index algnce2_idx on allegiance (code);
create index algnce3_idx on allegiance (name);

create table glossary (id int auto_increment not null, uri varchar(64) not null, title varchar(128) not null,
					   message text, primary key(id));
create index glossary1_idx on glossary (uri);


create table subsector(id int not null auto_increment, sector_id int not null,
                       idx int not null, name varchar(64), primary key(id));

create table map (planet_id int not null, image longblob not null, primary key(planet_id));
create table globe (planet_id int not null, image longblob not null, primary key(planet_id));

create table note (id int auto_increment not null, planet_id int not null, property varchar(16), message text, primary key(id));
